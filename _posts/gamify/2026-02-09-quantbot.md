      <details class="tutorial-glossary">
        <summary>Glossary (quick definitions)</summary>
        <div class="tutorial-glossary-grid">
          <div><strong>Lookback</strong>: how many past days the model can “see.”</div>
          <div><strong>Horizon</strong>: how many days ahead the model tries to predict.</div>
          <div><strong>Features</strong>: inputs (price/indicators/news signals) used to predict.</div>
          <div><strong>Label</strong>: what you’re trying to predict (future price or direction).</div>
          <div><strong>Train/Test split</strong>: hold-out data to evaluate generalization.</div>
          <div><strong>Hyperparameters</strong>: settings like lookback/horizon and model type.</div>
          <div><strong>Overfitting</strong>: learning noise instead of patterns.</div>
          <div><strong>Generalization</strong>: performance on unseen time periods.</div>
          <div><strong>Leakage</strong>: accidentally using future info in training.</div>
          <div><strong>Directional accuracy</strong>: makes sense only if the label is “up/down.”</div>
          <div><strong>MAE/RMSE</strong>: average / squared error in dollars (lower is better).</div>
          <div><strong>R²</strong>: explained variance (can be negative).</div>
          <div><strong>Random Forest</strong>: an ensemble of trees that reduces variance.</div>
          <div><strong>Linear Regression</strong>: interpretable baseline (often underfits).</div>
          <div><strong>LSTM</strong>: sequence model (stub in this demo).</div>
          <div><strong>Signal</strong>: model output transformed into a trading rule.</div>
        </div>
      </details>

  function formatTimeAgo(date) {
    if (!date) return "Last updated --";
    const now = Date.now();
    const diffMs = now - date.getTime();
    const diffSec = Math.floor(diffMs / 1000);
    if (diffSec < 60) return "Last updated just now";
    const diffMin = Math.floor(diffSec / 60);
    if (diffMin < 60) {
      return `Last updated ${diffMin} min${diffMin === 1 ? "" : "s"} ago`;
    }
    const diffHr = Math.floor(diffMin / 60);
    return `Last updated ${diffHr} hr${diffHr === 1 ? "" : "s"} ago`;

  // Accept Instant strings or YYYY-MM-DD strings
  function toYMD(timeVal) {
    if (!timeVal) return "";
    if (typeof timeVal === "string" && timeVal.length >= 10 && timeVal[4] === "-" && timeVal[7] === "-") {
      return timeVal.slice(0, 10);
    }
    try {
      return new Date(timeVal).toISOString().split("T")[0];
    } catch {
      return String(timeVal).slice(0, 10);
    }

      vol.push(Number(b.volume));

          <td>${h.qty ?? ""}</td>
          <td>$${Number(h.avgPrice ?? h.avgCost ?? 0).toFixed(2)}</td>
          <td>$${Number(h.lastPrice ?? h.marketPrice ?? 0).toFixed(2)}</td>
          <td style="color:${pnlColor}; font-weight:600;">${pnl >= 0 ? '+' : ''}$${pnl.toFixed(2)}</td>
        </tr>`;
      }
      html += `</tbody></table>`;
      holdingsDiv.innerHTML = html;
    } else if (!Array.isArray(holdings) && holdings && Object.keys(holdings).length > 0) {
      let html = `<table class="miniTable">
        <thead>
          <tr>
            <th>Ticker</th>
            <th>Quantity</th>
            <th>Avg Price</th>
            <th>Last Price</th>
          </tr>
        </thead>
        <tbody>`;
      
      for (const k of Object.keys(holdings)) {
        const h = holdings[k];
        html += `<tr>
          <td style="font-weight:600;">${h.ticker ?? k}</td>
          <td>${h.qty ?? ""}</td>
          <td>$${Number(h.avgPrice ?? h.avgCost ?? 0).toFixed(2)}</td>
          <td>$${Number(h.lastPrice ?? h.marketPrice ?? 0).toFixed(2)}</td>
        </tr>`;
      }
      html += `</tbody></table>`;
      holdingsDiv.innerHTML = html;
    } else {
      holdingsDiv.innerHTML = '<div style="color:var(--text-secondary); text-align:center; padding:20px;">No holdings yet</div>';
    }

    setStatus("Portfolio loaded");
  }

  async function placePaperOrder() {
    const personIdRaw = document.getElementById("personIdInput").value;
    const personId = personIdRaw ? parseInt(personIdRaw) : null;
    const ticker = document.getElementById("tickerInput").value.trim().toUpperCase();
    const side = document.getElementById("orderSide").value;
    const qty = parseInt(document.getElementById("orderQty").value);

    if (!personId) {
      setStatus("Enter Person ID first");
      return;
    }
    if (!ticker) {
      setStatus("Enter ticker");
      return;
    }
    if (!qty || qty <= 0) {
      setStatus("Quantity must be > 0");
      return;
    }

    const payload = {
      personId: personId,
      ticker: ticker,
      side: side,
      qty: qty,
      type: "market"
    };

    setStatus("Executing order...");
    const result = await apiPost("/bank/quant/paper/order", payload);

    if (result && result.success === false) {
      setStatus(result.note || "Order failed");
      return;
    }

    window.quantState.hasPaperTrade = true;
    setStatus(result.note || "Order executed");
    await loadPortfolio();
  }
